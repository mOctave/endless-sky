test-data "Npc Destination Mission"
	category mission
	contents
		mission "npc"
			landing
			npc evade
				ship "Quarg Wardragon"
				government "Quarg"
				destination "Over the Rainbow"
				personality uninterested


test "Npc Destination Test"
	status active
	description "Starts the mission, accepts it and then boards the quarg ship to capture it."
	sequence
		inject "Npc Destination Mission"
		inject "Npc Destination Save"
		call "Load First Savegame"
		watchdog 12000
		call "Depart"
		call "Land"
		# trigger the mission
		call "Depart"
		# empty input to make time pass whilst the quarg ship gets away - and stay there
		apply
			"test: steps to wait" = 7 * 60 + 30
		label "floating"
		# empty input to make time pass
		input
		apply
			"test: steps to wait" = "test: steps to wait" - 1
		branch "floating"
			"test: steps to wait" > 0
		call "Land"
		call "Wait 10 steps"
		assert
			"npc: offered" == 1
			"npc: done" == 1
